# function for calcutating the requrired statistics using 'groupby' and 'agg'
def calc_stats(group_col):
    stats = df.groupby(group_col).agg({
    'runs': 'sum',
    'wkts': 'sum',
    'runs_per_ball': 'mean',
    'econ': 'mean',
    '4s': 'sum',
    '6s': 'sum',
    'sr': 'mean'
    }).reset_index()
    return stats
